# Variáveis de ambiente do Supabase
# Copie este arquivo para .env e preencha com suas credenciais

# URL do seu projeto Supabase
VITE_SUPABASE_URL=https://seu-projeto.supabase.co

# Chave pública (anon key) do Supabase
VITE_SUPABASE_PUBLISHABLE_KEY=sua-chave-publica-aqui

# IMPORTANTE: 
# - Nunca commite o arquivo .env no Git
# - Use apenas a chave pública (anon key), nunca a service role key
# - Essas variáveis são expostas no frontend, então são seguras para uso público
